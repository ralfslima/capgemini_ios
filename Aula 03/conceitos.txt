/*TIPOS DE DADOS
 String -> Textual
 Char   -> Letra/Caractere
 
 Int    -> Números inteiros
 Float  -> Suporte entre 6 e 7 casas decimais
 Double -> Double entre 15 e 16 casas decimais
 UInt   -> Não aceita números negativos
 
 Bool   -> Boolean/Lógica (true / false)
 
 Any       -> Flexível
 AnyObject -> Classes
 
 Void   -> Quando não há retorno em funções
*/

/* OPERADORES ARITMÉTICOS
    +  -> SOMA
    -  -> SUBTRAÇÃO
    *  -> MULTIPLICAÇÃO
    /  -> DIVISÃO
    %  -> MÓDULO (RESTO DA DIVISÃO)
 
   OPERADORES RELACIONAIS
   >   -> MAIOR
   >=  -> MAIOR OU IGUAL
   <   -> MENOR
   <=  -> MENOR OU IGUAL
   ==  -> IGUAL
   !=  -> DIFERENTE
 
   OPERADORES LÓGICOS
   &&  -> E
   ||  -> OU
   !   -> NEGAÇÃO
 */
