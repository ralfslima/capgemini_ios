// 3º Exemplo
import Foundation

func tarefa1() async {
    try? await Task.sleep(for:.seconds(2))
    print("Tarefa 1")
}

func tarefa2() async {
    //Thread.sleep(forTimeInterval: 5)
    try? await Task.sleep(for:.seconds(5))
    print("Tarefa 2")
}

func tarefa3() async {
    //Thread.sleep(forTimeInterval: 3)
    try? await Task.sleep(for:.seconds(3))
    print("Tarefa 3")
}

// Tarefa principal
Task {
    // Agrupar tarefas
    await withTaskGroup(of: Void.self) { grupo in
        
        // Adicionar a primeira tarefa
        grupo.addTask { await tarefa1() }
            
        // Adicionar a segunda tarefa
        grupo.addTask { await tarefa2() }
        
        // Adicionar a terceira tarefa
        grupo.addTask { await tarefa3() }
        
    }
    
    // Retorno informado o término das tarefas
    print("Tarefas finalizadas")
}
