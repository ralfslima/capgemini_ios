/*LISTA 04 - PROTOCOLS, GENERICS e STRUCTS*/
protocol Conta {
    var id: Int { get }
    var nome: String { get }
    var saldo: Double { get set }
}

struct ContaPF: Conta {
    let id: Int
    var nome: String
    var saldo: Double
    var cpf: String
}

struct ContaPJ: Conta {
    let id: Int
    var nome: String
    var saldo: Double
    var cnpj: String
}

struct ContaCRUD<T: Conta> {

    private var lista: [T] = []

    mutating func cadastrar(conta: T)  -> Void {
        lista.append(conta)
        print("✅ Conta cadastrada: \(conta.nome)")
    }

    func listar() -> [T] {
        return lista
    }

    mutating func alterar(id: Int, novaConta: T)  -> Void {
        if let index = lista.firstIndex(where: { $0.id == id }) {
            lista[index] = novaConta
            print("✏️ Conta alterada com sucesso!")
        } else {
            print("❌ Conta não encontrada.")
        }
    }

    mutating func remover(id: Int) -> Void {
        lista.removeAll { $0.id == id }
        print("🗑️ Conta removida.")
    }
}

var crudPF = ContaCRUD<ContaPF>()
var crudPJ = ContaCRUD<ContaPJ>()

// Criando contas
let pf1 = ContaPF(id: 1, nome: "Felipe", saldo: 1000, cpf: "123")
let pj1 = ContaPJ(id: 1, nome: "Empresa X", saldo: 5000, cnpj: "999")

// Cadastrar
crudPF.cadastrar(conta: pf1)
crudPJ.cadastrar(conta: pj1)

// Listar
print("\n📋 Contas PF:")
crudPF.listar().forEach {
    print("\($0.nome) | CPF: \($0.cpf) | Saldo: \($0.saldo)")
}

print("\n📋 Contas PJ:")
crudPJ.listar().forEach {
    print("\($0.nome) | CNPJ: \($0.cnpj) | Saldo: \($0.saldo)")
}

// Alterar
let pfAtualizado = ContaPF(id: 1, nome: "Felipe Silva", saldo: 2000, cpf: "123")
crudPF.alterar(id: 1, novaConta: pfAtualizado)

// Remover
crudPJ.remover(id: 1)
