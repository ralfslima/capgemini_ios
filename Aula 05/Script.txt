// FUNÇÕES PARA MANIPULAR VETORES
var idades: [Int] = [12, 32, 8, 41, 33, 16, 27]

print("A quantidade de registros é \(idades.count)")

print("O array está vazio? Resposta: \(idades.isEmpty)")

print(idades.first ?? "Vetor vazio")

print(idades.first(where: {$0 > 20} ) ?? "Não existe idade maior que 20")

print(idades.last ?? "Vetor vazio")

print(idades.filter { $0 >= 18 })

print(idades.sorted())

print(idades.sorted(by: >))

print(idades.contains(32))

for idade in idades where idade >= 18 {
    print(idade)
}

let situacaoIdade = idades.map {
    $0 >= 18 ? "\($0) é maior de idade" : "\($0) é menor de idade"
}

situacaoIdade.forEach { print($0) }

// FUNÇÕES PARA MANIPULAR TEXTOS
let texto: String = "Aprendendo Swift"

print(Array(texto)[0]) // Extrair caractere em uma determinada posição

print(texto.count) // Obter a quantidade de caracteres

print(texto.isEmpty) // Retorna um tipo lógico, dependendo se o vetor possui alguma informação

print(texto.uppercased()) // Transforma os caracteres em "caixa alta" (maiúsculas)

print(texto.lowercased()) // Transforma os caracteres em "caixa baixa" (minúsculas)

let nomes: String = "Alice, Carlos, Roberto, Pâmela, Alice" // Nomes separados por vírgula

let vetorNomes = nomes.split(separator: ", ") // Criar um novo vetor, será gerada um novo valor, quando referenciar a vírgula

print(vetorNomes) // Exibir vetor de nomes

print(nomes.replacing("Alice", with: "Aline")) // Substituir informações
